# Commands used in the e-commerce project

## Install dependencies
py -m pip install -r requirements.txt

## Create database tables
python -c "from database import Base, engine; Base.metadata.create_all(bind=engine)"

## Insert sample data
python sample_data.py

## Run the FastAPI application locally
py -m uvicorn main:app --reload

## Test endpoints (examples)
curl -X GET "http://localhost:8000/products"
curl -X POST "http://localhost:8000/cart" -H "Content-Type: application/json" -H "Authorization: Bearer YOUR_API_KEY" -d '{"product_id": 1, "quantity": 2}'
curl -X GET "http://localhost:8000/cart" -H "Authorization: Bearer YOUR_API_KEY"
curl -X POST "http://localhost:8000/checkout" -H "Content-Type: application/json" -H "Authorization: Bearer YOUR_API_KEY" -d '{"product_id": 1, "quantity": 1}'
curl -X GET "http://localhost:8000/orders" -H "Authorization: Bearer YOUR_API_KEY"
curl -X PUT "http://localhost:8000/cart/1" -H "Content-Type: application/json" -H "Authorization: Bearer YOUR_API_KEY" -d '{"quantity": 3}'
curl -X DELETE "http://localhost:8000/cart/1" -H "Authorization: Bearer YOUR_API_KEY"
